% draw text to absolute coordinates
\newcommand{\txt}[3]{\begin{textblock*}{\panelWidth}(#1\panelWidth+\txtXshift,#2\panelHeight+\txtYshift){#3}\end{textblock*}}
% draw label
\newcommand{\panelNumber}[3]{\begin{textblock*}{\panelWidth}(#1\panelWidth+\panelNumberXshift,#2\panelHeight+\panelNumberYshift)\textbf{\LARGE #3}\end{textblock*}}
% draw title: \panelTitle{x}{y}{text}
\newcommand{\panelTitle}[3]{\begin{textblock*}{\panelWidth}(#1\panelWidth+\panelTitleXshift,#2\panelHeight+\panelTitleYshift)\textbf{\large #3}\end{textblock*}}
% draw row title: \rowTitle{y}{text}
\newcommand{\rowTitle}[2]{\begin{textblock*}{\panelWidth}(\rowTitleXshift,#1\panelHeight+\rowTitleYshift){\rotatebox{90}{\textbf{\large #2}}}\end{textblock*}}

% x-Axis label
\newcommand{\xLab}[3]{\begin{textblock*}{\panelWidth}(#1\panelWidth+\xLabXshift,#2\panelHeight+\xLabYshift){\large #3}\end{textblock*}}
% y-Axis label
\newcommand{\yLab}[3]{\begin{textblock*}{\panelWidth}(#1\panelWidth+\yLabXshift,#2\panelHeight+\yLabYshift){\rotatebox{90}{\large #3}}\end{textblock*}}

% draw graphics: 
% \pic{x}{y}{basename of a pdf file}
\newcommand{\pic}[3]{\begin{textblock*}{\panelWidth}(#1\panelWidth+\picXshift,#2\panelHeight+\picYshift)\IfFileExists{#3.pdf}{\includegraphics[width=\picWidth]{{#3}.pdf}}{}\end{textblock*}}
% \widepic{x}{y}{basename of a pdf file}{picture width}
\newcommand{\widepic}[4]{\begin{textblock*}{\panelWidth}(#1\panelWidth+\picXshift,#2\panelHeight+\picYshift)\IfFileExists{#3.pdf}{\includegraphics[width=#4]{{#3}.pdf}}{}\end{textblock*}}

% draw labelled subfigure
\newcommand{\fig}[5]{ \panelNumber{#1}{#2}{#3} \pic{#1}{#2}{#4} \panelTitle{#1}{#2}{#5} }
\newcommand{\figVarWidth}[6]{ \panelNumber{#1}{#2}{#3} \widepic{#1}{#2}{#4}{#6} \panelTitle{#1}{#2}{#5} }
